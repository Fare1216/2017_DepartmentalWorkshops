#include "DxLib.h"


int Key[256];

int x = 320; //x座標
int y = 240; //y座標

int Charcter[2];

int Time = 0;
int mode = 0;


int gpUpdateKey(){
	char tmpKey[256];
	GetHitKeyStateAll( tmpKey );

	for(int i=0;i<256;i++){
		if(tmpKey[i] != 0){
			Key[i]++;
		}
		else{
			Key[i] = 0;
		}
	}

	return 0;
}

int WINAPI WinMain(HINSTANCE,HINSTANCE,LPSTR,int){
 
	ChangeWindowMode(TRUE), DxLib_Init(), SetDrawScreen( DX_SCREEN_BACK );

	LoadDivGraph("charcter.png", 2, 2, 1, 32, 32, Charcter);


	while( ScreenFlip()==0 && ProcessMessage()==0 && ClearDrawScreen()==0 && gpUpdateKey()==0 ){
		if( Key[ KEY_INPUT_RIGHT ] >= 1){
			if(x+32 < 639){
				x+=2;
			}
		}
		if( Key[ KEY_INPUT_DOWN ] >= 1){
			if(y+32 < 479){
				y+=2;
			}
		}
		if( Key[ KEY_INPUT_LEFT ] >= 1){
			if(x > 0){
				x-=2;
			}
		}
		if( Key[ KEY_INPUT_UP ] >= 1){
			if(y > 0){
				y-=2;
			}
		}


		Time++;

		if(Time == 20){
			mode =(mode+1)%2;
			Time = 0;
		}

		switch(mode){
		case 0:{
			DrawGraph(x, y, Charcter[0], TRUE);
			break;
			   }
		case 1:{
			DrawGraph(x, y, Charcter[1], TRUE);
			break;
			   }
		}
		
		ScreenFlip();
	}

	DxLib_End(); // DXライブラリ終了処理
	return 0;

}
